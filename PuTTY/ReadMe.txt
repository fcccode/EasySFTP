- PUTTYMEM.H
- PUTTYPS.H
- SSH.H
- SSHBN.C
- TREE234.H
- WINDOWS\WINHELP.H  (expand to .\WINDOWS directory)
- WINDOWS\WINPGNTC.C (expand to .\WINDOWS directory)
- WINDOWS\WINSTUFF.H (expand to .\WINDOWS directory)
